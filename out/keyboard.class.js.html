/**
 * Represents the state of keyboard inputs. Each property corresponds
 * to a particular key and holds a boolean indicating whether that key
 * is currently pressed.
 */
class Keyboard {
    LEFT = false;
    RIGHT = false;
    UP = false;
    DOWN = false;
    SPACE = false;
    RSHIFT = false;
}
